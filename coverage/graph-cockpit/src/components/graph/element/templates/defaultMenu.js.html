/** Template for the Default context menu.
@param {function} deleteItem: Deletes the selected item.

	return [
		{
			label: 'Associate To',
			// accelerator: 'CmdOrCtrl+F',
			click(menuItem, focusedWindow, event){
				console.log("Associate To");
